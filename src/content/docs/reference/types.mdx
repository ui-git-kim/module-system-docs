## Registry Types

### RegistryItem

```typescript
interface RegistryItem {
  id: string;
  order?: number;
}
```

### NavigationItem

```typescript
interface NavigationItem extends RegistryItem {
  id: string;
  title: string;
  url: string;
  icon?: LucideIcon;
  section: 'main' | 'user';
  items?: { title: string; url: string }[];
  requiredRole?: string | string[];
}
```

### DashboardWidget

```typescript
interface DashboardWidget extends RegistryItem {
  id: string;
  title: string;
  component: ComponentType<DashboardWidgetProps>;
  size: '1x1' | '2x1' | '1x2' | '2x2' | 'full';
  category?: string;
  canHide?: boolean;
}

interface DashboardWidgetProps {
  widget: DashboardWidget;
  isCompact?: boolean;
}
```

### SettingsPage

```typescript
interface SettingsPage extends RegistryItem {
  id: string;
  title: string;
  description?: string;
  icon?: LucideIcon;
  component: ComponentType;
  isReplacement?: boolean;
}
```

### HeaderToolbarItem

```typescript
interface HeaderToolbarItem extends RegistryItem {
  id: string;
  type: 'action' | 'toggle' | 'dropdown' | 'custom';
  position: 'left' | 'right';
  icon?: LucideIcon;
  label?: string;
  onClick?: () => void;
  isActive?: boolean;
  badge?: number;
  component?: ComponentType;
}
```

## Filter & Action Types

```typescript
type FilterCallback<T, C> = (value: T, context: C) => T;
type AsyncFilterCallback<T, C> = (value: T, context: C) => Promise<T>;

type ActionCallback<C> = (context: C) => void;
type AsyncActionCallback<C> = (context: C) => Promise<void>;
```
